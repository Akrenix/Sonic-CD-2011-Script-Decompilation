sub ObjectMain
	switch Object.State
	case 0
		if Object.Value0<8
			Object.Value0++
		else
			Object.Value0=254
			Object.State++
		endif
		SetScreenFade(0,0,0,255)
		break
	case 1
		if Object.Value0>0
			SetScreenFade(0,0,0,Object.Value0)
			Object.Value0-=8
		else
			Object.State++
		endif
		break
	case 2
		if KeyPress.Up==1
			Object.Value2=0
			Object.Value1--
			if Object.Value1<0
				Object.Value1=69
			endif
		else
			if KeyPress.Down==1
				Object.Value2=0
				Object.Value1++
				if Object.Value1>69
					Object.Value1=0
				endif
			endif
		endif
		if KeyDown.Up==1
			Object.Value2++
			if Object.Value2==8
				Object.Value2=0
				Object.Value1--
				if Object.Value1<0
					Object.Value1=69
				endif
			endif
		else
			if KeyDown.Down==1
				Object.Value2++
				if Object.Value2==8
					Object.Value2=0
					Object.Value1++
					if Object.Value1>69
						Object.Value1=0
					endif
				endif
			endif
		endif
		if KeyPress.Start==1
			Object.State=3
		endif
		if KeyPress.ButtonB==1
			Object.State=4
		endif
		break
	case 3
		if Object.Value0<288
			Object.Value0+=8
		else
			Options.GameMode=0
			Stage.ActiveList=REGULAR_STAGE
			Stage.ListPos=Object.Value1
			LoadStage()
		endif
		TempValue0=Object.Value0
		if TempValue0>255
			TempValue0=255
		endif
		SetScreenFade(0,0,0,TempValue0)
		break
	case 4
		if Object.Value0<288
			Object.Value0+=8
		else
			Options.GameMode=0
			Stage.ActiveList=PRESENTATION_STAGE
			Stage.ListPos=1
			LoadStage()
		endif
		TempValue0=Object.Value0
		if TempValue0>255
			TempValue0=255
		endif
		SetScreenFade(0,0,0,TempValue0)
		break

	endswitch
endsub


sub ObjectDraw
	TempValue0=0
	while TempValue0<240
		TempValue1=0
		while TempValue1<Screen.XSize
			DrawSpriteScreenXY(15,TempValue1,TempValue0)
			TempValue1+=224
		loop
		TempValue0+=48
	loop
	DrawSpriteScreenXY(16,Screen.CenterX,80)
	TempValue0=Screen.CenterX
	TempValue0-=48
	TempValue1=Object.Value1
	TempValue1/=10
	switch TempValue1
	case 0
		DrawSpriteScreenXY(1,TempValue0,112)
		break
	case 1
		DrawSpriteScreenXY(3,TempValue0,112)
		break
	case 2

		break
	case 3
		DrawSpriteScreenXY(5,TempValue0,112)
		break
	case 4
		DrawSpriteScreenXY(6,TempValue0,112)
		break
	case 5
		DrawSpriteScreenXY(7,TempValue0,112)
		break
	case 6
		DrawSpriteScreenXY(8,TempValue0,112)
		break

	endswitch
	TempValue0+=8
	DrawSpriteScreenXY(14,TempValue0,112)
	TempValue0+=8
	TempValue1=Object.Value1
	TempValue1%=10
	TempValue1>>=2
	TempValue1++
	DrawSpriteScreenXY(TempValue1,TempValue0,112)
	TempValue0+=8
	DrawSpriteScreenXY(14,TempValue0,112)
	TempValue0+=8
	TempValue1=Object.Value1
	TempValue1%=10

	case 4

	case 3

	case 9

		break

	endswitch
endsub
